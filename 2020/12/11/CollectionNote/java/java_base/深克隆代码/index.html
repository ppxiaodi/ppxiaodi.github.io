package com.example.act.kabanda;

import lombok.Getter;
import lombok.Setter;

import java.io.*;

@Getter
@Setter
public class OutObject implements Cloneable,Serializable{

    private String book;

  private InnerObject innerObject;



    @Override
    protected OutObject clone() throws CloneNotSupportedException {
        OutObject outObject = (OutObject)super.clone();
        InnerObject innerObject2 = innerObject.clone();
        outObject.setInnerObject(innerObject2);
        return outObject;
    }


}

class Test1{
    public static void main(String[] args) throws CloneNotSupportedException, IOException, ClassNotFoundException {
        OutObject outObject = new OutObject();
        String book = "123";
        InnerObject innerObject = new InnerObject("qbc","qer");
        outObject.setBook(book);
        outObject.setInnerObject(innerObject);


        OutObject outObject2 = outObject.clone();
        System.out.println(outObject.hashCode()+">>>>"+ outObject.getInnerObject().hashCode());
        System.out.println(outObject2.hashCode()+">>>>"+ outObject2.getInnerObject().hashCode());

       ByteArrayOutputStream bt = new ByteArrayOutputStream();
       ObjectOutputStream ob = new ObjectOutputStream(bt);
       ob.writeObject(outObject);
        ob.close();
       ObjectInputStream io = new ObjectInputStream(new ByteArrayInputStream(bt.toByteArray()));
        OutObject outObject3 = (OutObject) io.readObject();
        System.out.println(outObject3.hashCode()+">>>>"+ outObject3.getInnerObject().hashCode());

    }
}

package com.example.act.kabanda;


import lombok.Getter;
import lombok.Setter;

import java.io.Serializable;

@Getter
@Setter
public class InnerObject implements Cloneable, Serializable {
    public InnerObject(String type, String name) {
        this.type = type;
        this.name = name;
    }

    String type;
    String name;


    @Override
    protected InnerObject clone() throws CloneNotSupportedException {
        return (InnerObject)super.clone();
    }
}
