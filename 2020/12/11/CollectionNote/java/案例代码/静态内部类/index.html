        OuterCase.InnerStaticCase innerStaticCase = new OuterCase.InnerStaticCase();

        innerCase.print();

        innerStaticCase.print();
    }
}


 class OuterCase {

    private static String sex;

    private String name;

    public String getName() {
        return name;
    }

    public static String strValue() {
        return "1231";
    }

    public void setName(String name) {
        this.name = name;
    }

    static class InnerStaticCase{
        public void print(){
            System.out.println(OuterCase.sex);
        }
    }

    class InnerCase{
        public void print(){
            System.out.println(OuterCase.sex);
            System.out.println(OuterCase.this.name);
        }
    }
}
