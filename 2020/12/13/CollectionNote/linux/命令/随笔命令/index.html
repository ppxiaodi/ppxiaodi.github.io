vi Dockerfile
FROM java:8
EXPOSE 9001
COPY user-1.0-SNAPSHOT.jar /mydata/
RUN mkdir /APP && mv /mydata/user-1.0-SNAPSHOT.jar /APP/user.jar
#ENTRYPOINT ["nohup","java", "-jar","/mydata/user-1.0-SNAPSHOT.jar","&"]
#ADD user-1.0-SNAPSHOT.jar /user.jar
ENTRYPOINT ["nohup","java", "-jar","/APP/user.jar","&"]
docker build -t user .
docker run  -p 9002:9001 --name myuser  -d user
docker logs myuser
docker exec -it myuser /bin/bash
