@GetMapping(value="/template")
   @ApiOperation(value = "下载导入模板",response = Result.class)
   public void downTemplate(HttpServletResponse response) {
       InputStream inputStream = null;
       OutputStream outputStream = null;
       BufferedInputStream bis = null;
       try {
           String fileName = "xxx导入模板.xls";
           // 配置文件下载
           response.setHeader("content-type", "application/octet-stream");
           response.setContentType("application/octet-stream");
           // 下载文件能正常显示中文
           response.setHeader("Content-Disposition", "attachment;filename=" + URLEncoder.encode(fileName, "UTF-8"));

           ClassPathResource classPathResource = new ClassPathResource("excelTemplate/student/"+fileName);
           //实现文件下载
           byte[] buffer = new byte[1024];
           inputStream =classPathResource.getInputStream();
           bis = new BufferedInputStream(inputStream);
           outputStream =  response.getOutputStream();
           int i = bis.read(buffer);
           while (i != -1) {
               outputStream.write(buffer, 0, i);
               i = bis.read(buffer);
           }
       } catch (Exception e) {
           log.error("下载xxx导入模板失败", e);
       }finally {
           if(bis!=null){
               try {
                   bis.close();
               } catch (IOException e) {
                   log.error("关闭流失败", e);
               }
           }
           if(inputStream!=null){
               try {
                   inputStream.close();
               } catch (IOException e) {
                   log.error("关闭流失败", e);
               }
           }
           if(outputStream!=null){
               try {
                   outputStream.close();
               } catch (IOException e) {
                   log.error("关闭流失败", e);
               }
           }
       }
   }

import org.apache.poi.hssf.usermodel.HSSFWorkbook;
import org.apache.poi.ss.usermodel.Row;
import org.apache.poi.ss.usermodel.Sheet;
import org.apache.poi.ss.usermodel.Workbook;
import org.apache.poi.xssf.usermodel.XSSFWorkbook;

    public Result importByExcel(InputStream inputStream,String extName) throws Exception {
        Long userId = CurrentUserUtil.getUserId();
        String orgId = CurrentUserUtil.getOrgId();
        Date date = new Date();

        List<StudentEthnicMinority> list = new ArrayList<>();
        Workbook workbook;
        Sheet sheet;
        if("xlsx".equalsIgnoreCase(extName)){
            workbook = new XSSFWorkbook(inputStream);
            sheet = workbook.getSheetAt(0);
        }else{
            workbook = new HSSFWorkbook(inputStream);
            sheet = workbook.getSheetAt(0);
        }


        //获取行数
        int lastRowNum = sheet.getLastRowNum();
        if(lastRowNum<3){
            return Result.error("请添加数据");
        }
        if(lastRowNum > 202){
            return Result.error("批量导入超过200人的学生");
        }
        List<ExcelImportErrorMessage> excelImportErrorMessageList = new ArrayList<>();
        List<ImportEncodeIdcard> importEncodeIdcardList = new ArrayList<>();
        int id = 0;
        for (int i = 3; i <= lastRowNum; i++) {
            StringBuffer errMassage = new StringBuffer();
            StringBuffer userName = new StringBuffer();
            Row row = sheet.getRow(i);
            StudentEthnicMinority studentEthnicMinority = new StudentEthnicMinority();
            studentEthnicMinority.setCreateId(userId);
            studentEthnicMinority.setGmtCreate(date);
            studentEthnicMinority.setModifiedId(userId);
            studentEthnicMinority.setGmtModified(date);
            studentEthnicMinority.setOrgId(orgId);
            studentEthnicMinority.setStatus((byte) 1);

            //姓名
            ExcelDataVerification.name( row, 0,  errMassage,  userName, studentEthnicMinority);

            //性别
            ExcelDataVerification.gender( row, 1,  errMassage,  studentEthnicMinority);

            //年级
            ExcelDataVerification.grade( row, 2,  errMassage,  studentEthnicMinority);

            //班级
            ExcelDataVerification.className( row, 3,  errMassage,  studentEthnicMinority);

            //民族
            ExcelDataVerification.nation( row, 4,  errMassage,  studentEthnicMinority);

            //身份证号码
            ExcelDataVerification.idCard( row, 5,i,  errMassage, importEncodeIdcardList, studentEthnicMinority);

            //班主任
            ExcelDataVerification.headmaster( row, 6,  errMassage,  studentEthnicMinority);

            //存在错误信息
            if(!StringUtils.isBlank(errMassage)){
                ExcelImportErrorMessage excelImportErrorMessage = new ExcelImportErrorMessage();
                excelImportErrorMessage.setId(++id);
                excelImportErrorMessage.setName(userName.toString());
                String line = "第"+(i + 1)+"行";
                excelImportErrorMessage.setLineNumber(line);
                excelImportErrorMessage.setErrMessage(errMassage.substring(0,errMassage.length()-1));

                excelImportErrorMessageList.add(excelImportErrorMessage);
            }
            list.add(studentEthnicMinority);
        }
        
        if(list.size()!=0) {
            batchInsert(list);
        }
        return Result.ok();
    }

import org.apache.commons.lang3.StringUtils;
import org.apache.poi.ss.usermodel.Cell;

public class ExcelDataVerification {

    //姓名
    public static void name(Row row, int column, StringBuffer errMassage, StringBuffer userName, StudentBaseInfo studentBaseInfo) {
        // if (row.getCell(0) != null) {
        if (row!=null && row.getCell(column) != null) {
            Cell cell = row.getCell(column);
            cell.setCellType(CellType.STRING);
            String trueName = cell.getStringCellValue();
            userName.append(trueName);
            if (StringUtils.isBlank(trueName)) {
                //return Result.error("第" + (i + 1) + "行,第" + 1 + "列,为空");
                errMassage.append("姓名不能为空,");
            } else if (!RegexUtils.regexWord(trueName)) {
                // return Result.error("第" + (i + 1) + "行,第" + 1 + "列,有空格、符号");
                errMassage.append("姓名有空格、符号,");
            } else {
                studentBaseInfo.setTrueName(trueName);
            }
        } else {
            //return Result.error("第" + (i + 1) + "行,第" + 1 + "列,为空");
            errMassage.append("姓名不能为空,");
        }
    }

    //性别
    public static void gender(Row row, int column, StringBuffer errMassage, StudentBaseInfo studentBaseInfo) {
        //if (row.getCell(1) != null) {
        if (row!=null && row.getCell(column) != null) {
            Cell cell = row.getCell(column);
            cell.setCellType(CellType.STRING);
            String genderStr = cell.getStringCellValue();
            if (StringUtils.isBlank(genderStr)) {
                // return Result.error("第" + (i + 1) + "行,第" + 2 + "列,为空");
                errMassage.append("性别不能为空,");
            } else {
                genderStr = genderStr.replaceAll("\\s*", "");
                byte gender = Gender.getValueByName(genderStr);
                if (gender == -1) {
                    // return Result.error("第" + (i + 1) + "行,第" + 2 + "列,格式不对");
                    errMassage.append( "性别格式不对,");
                } else {
                    studentBaseInfo.setGender(gender);
                }

            }
        } else {
            // return Result.error("第" + (i + 1) + "行,第" + 2 + "列,为空");
            errMassage.append( "性别不能为空,");
        }
    }

    //民族
    public static void nation(Row row, int column, StringBuffer errMassage, StudentEthnicMinority studentEthnicMinority) {
        //if (row.getCell(4) != null) {
        if (row!=null && row.getCell(column) != null) {
            Cell cell = row.getCell(column);
            cell.setCellType(CellType.STRING);
            String nation = cell.getStringCellValue();
            if (StringUtils.isBlank(nation)) {
                // return Result.error("第" + (i + 1) + "行,第" + 5 + "列,为空");
                errMassage.append( "民族不能为空,");
            } else if (!RegexUtils.regexChineseWord(nation)) {
                //return Result.error("第" + (i+1) + "行,第" + 5 + "列,格式不对");
                errMassage.append( "民族格式不对,");
            } else {
                nation = nation.replaceAll("\\s*", "");
                if (!nation.contains("族")) {
                    // return Result.error("第" + (i+1) + "行,第" + 5 + "列,格式不对");
                    errMassage.append("民族格式不对,");
                } else {
                    studentEthnicMinority.setNation(nation);
                }
            }
        } else {
            //  return Result.error("第" + (i+1) + "行,第" + 5 + "列,为空");
            errMassage.append( "民族不能为空,");
        }
    }

    public static void idCard(Row row, int column,int excelRow, StringBuffer errMassage, List<ImportEncodeIdcard> importEncodeIdcardList, StudentBaseInfo studentBaseInfo) throws Exception {

                // return Result.error("第" + (i + 1) + "行,第" + 7 + "列,为空");
                errMassage.append( "证件号不能为空,");
            } else {
                idCard = idCard.replaceAll("\\s*", "");
                String encodeIdcard = DESCrypUtils.idCardEncrypt(idCard);
                studentBaseInfo.setEncodeIdcard(encodeIdcard);

                importEncodeIdcard.setEncodeIdcard(encodeIdcard);

            }
        } else {
            // return Result.error("第" + (i + 1) + "行,第" + 7 + "列,为空");
            errMassage.append( "证件号不能为空,");
        }
    }
