package com.example.spring_test;

import lombok.extern.log4j.Log4j2;
import org.springframework.boot.autoconfigure.data.redis.RedisProperties;
import redis.clients.jedis.*;

import java.util.HashSet;
import java.util.List;
import java.util.Set;
@Log4j2
public class RedisCacheUtil {

    private static JedisCommands jedisCommands;

    private static JedisPool jedisPool;

    public RedisCacheUtil(RedisProperties redisProperties) {
        if (isSingleMode(redisProperties)) {
            initialPool(redisProperties);
        } else {
            initialCluster(redisProperties);
        }
    }

    /**
     * 初始化单机版Redis连接池
     *
     * @param redisProperties redis配置参数
     */

    private void initialPool(RedisProperties redisProperties) {
        JedisPoolConfig config = getJedisPoolConfig(redisProperties);
        jedisPool = new JedisPool(config, redisProperties.getHost(), redisProperties.getPort(),(int) redisProperties.getTimeout().toMillis(),redisProperties.getPassword());
        //(final GenericObjectPoolConfig poolConfig, final String host, int port,
        //      int timeout, final String password)
    }

    /**
     * 初始化redis集群
     *
     * @param redisProperties redis配置参数
     */

    private void initialCluster(RedisProperties redisProperties) {
        Set<HostAndPort> jedisClusterNodes = new HashSet<>();
        List<String> nodeAddresses = redisProperties.getCluster().getNodes();
        for (String nodeAddress : nodeAddresses) {
            String[] hostAndPort = nodeAddress.split(":");
            if (hostAndPort.length < 2) {
                continue;
            }
            jedisClusterNodes.add(new HostAndPort(hostAndPort[0], Integer.parseInt(hostAndPort[1])));
        }
        JedisPoolConfig config = getJedisPoolConfig(redisProperties);
        jedisCommands = new JedisCluster(jedisClusterNodes,(int) redisProperties.getTimeout().toMillis(), 2000,5,redisProperties.getPassword(), config);

    }

    /**
     * 获取连接池配置对象
     *
     * @param redisProperties redis配置参数
     * @return 连接池配置对象

     */

    public static JedisPoolConfig getJedisPoolConfig(RedisProperties redisProperties) {
        JedisPoolConfig config = new JedisPoolConfig();
        RedisProperties.Pool pool = redisProperties.getJedis().getPool();
        config.setMaxTotal(pool.getMaxActive());
        config.setMaxIdle(pool.getMaxIdle());
        config.setMinIdle(pool.getMinIdle());
        config.setMaxWaitMillis(pool.getMaxWait().toMillis());
        config.setTestOnBorrow(true);
        return config;
    }

    /**
     * 判断redis是否是单机模式
     *
     * @return boolean

     */

    private boolean isSingleMode(RedisProperties redisProperties) {
        RedisProperties.Cluster cluster = redisProperties.getCluster();
        if (cluster == null) {
            return true;
        }
        List<String> nodes = cluster.getNodes();
        return nodes == null || nodes.isEmpty();
    }

    /**
     * 获取执行指令接口
     *
     * @return 指令接口
     */


    public static JedisCommands getJedisCommands() {
        if (jedisPool != null) {
            try {
                return jedisPool.getResource();
            } catch (Exception e) {
                log.error("系统异常",e);
            }
            return null;
        }
        return jedisCommands;
    }

    /**
     * 获取redis值
     * @param key key
     * @return value
     */


    public static String get(String key) {
        JedisCommands commands = getJedisCommands();
        if (commands != null) {
            try {
                return commands.get(key);
            } catch (Exception ex) {
                ex.printStackTrace();
            } finally {
                if (commands instanceof Jedis) {
                    ((Jedis) commands).close();
                }
            }
        }
        throw new NullPointerException("Instance of interface JedisCommands cannot be null.");
    }
}

package com.example.spring_test;

import org.springframework.beans.factory.annotation.Qualifier;
import org.springframework.boot.autoconfigure.data.redis.RedisProperties;
import org.springframework.boot.context.properties.ConfigurationProperties;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;

@Configuration
public class RedisConfigure {

    @Bean("remoteRedisProperties")
    @ConfigurationProperties(prefix = "auth.remote-redis")
    public RedisProperties remoteRedisProperties() {
        System.out.println("-------------------- remote auth redisProperties init ---------------------");
        return new RedisProperties();
    }

    @Bean(name = "redisCacheUtil")
    public RedisCacheUtil redisCacheUtil(@Qualifier("remoteRedisProperties") RedisProperties redisProperties) {
        return new RedisCacheUtil(redisProperties);
    }

     @Primary
    @Bean("redisProperties")
    @ConfigurationProperties(prefix = "spring.redis")
    public RedisProperties redisProperties() {
        System.out.println("-------------------- local auth redisProperties init ---------------------");
        return new RedisProperties();
    }


         @Bean
    public JedisCluster jedisCluster(@Qualifier("redisProperties") RedisProperties redisProperties) {
        Set<HostAndPort> jedisClusterNodes = new HashSet<>();
        List<String> nodeAddresses = redisProperties.getCluster().getNodes();
        for (String nodeAddress : nodeAddresses) {
            String[] hostAndPort = nodeAddress.split(":");
            if (hostAndPort.length < 2) {
                continue;
            }
            jedisClusterNodes.add(new HostAndPort(hostAndPort[0], Integer.parseInt(hostAndPort[1])));
        }
        JedisPoolConfig config = RedisCacheUtil.getJedisPoolConfig(redisProperties);
        return new JedisCluster(jedisClusterNodes,(int) redisProperties.getTimeout().toMillis(), 2000,5,redisProperties.getPassword(), config);
    }
    
}
