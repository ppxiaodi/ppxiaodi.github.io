import redis.clients.jedis.*;
import redis.clients.util.JedisClusterCRC16;

    @Autowired
    private JedisCluster jedisCluster;
 /**
     * 这个方法为一个槽中的批量删除
     * @param keys
     * @return
     */
    public Long delBatch(String... keys){
        return jedisCluster.del(keys);
    }
    @Override
    public Long del(String... keys) {
        if (keys == null || keys.length < 1) {
            return 0L;
        }
        if (keys.length == 1) {
            return jedisCluster.del(keys[0]);
        }
        long count = 0;
        Map<JedisPool, List<String>> poolKeysMap = getPoolKeysMap(keys);
        List<Object> result;
        for (Map.Entry<JedisPool, List<String>> entry : poolKeysMap.entrySet()) {
            try (Jedis jedis = entry.getKey().getResource()) {
                //count += jedis.del(entry.getValue().toArray(new String[0])); // 性能不如使用pipeline方式
                //获取pipeline
                Pipeline pipeline = jedis.pipelined();
                for (String key : entry.getValue()) {
                    pipeline.del(key);
                }
                //从pipeline中获取结果
                result = pipeline.syncAndReturnAll();
                pipeline.close();
                count += result.stream().filter(o -> o.equals(1L)).count();
            } catch (Exception e) {
                log.error("系统异常",e);
            }
        }
        return count;
    }
private Map<JedisPool, List<String>> getPoolKeysMap(String[] keys) {
        Map<JedisPool, List<String>> poolKeysMap = new LinkedHashMap<>(6);
        for (String key : keys) {
            JedisPool jedisPool;
            int slot = JedisClusterCRC16.getSlot(key);
            // JedisCluster继承了BinaryJedisCluster
            // BinaryJedisCluster的JedisClusterConnectionHandler属性
            // 里面有JedisClusterInfoCache，根据这一条继承链，可以获取到JedisClusterInfoCache
            // 从而获取slot和JedisPool直接的映射
            MetaObject metaObject = SystemMetaObject.forObject(jedisCluster);
            JedisClusterInfoCache cache = (JedisClusterInfoCache) metaObject.getValue("connectionHandler.cache");
            jedisPool = cache.getSlotPool(slot);
            if (poolKeysMap.containsKey(jedisPool)) {
                poolKeysMap.get(jedisPool).add(key);
            } else {
                List<String> subKeyList = new ArrayList<>();
                subKeyList.add(key);
                poolKeysMap.put(jedisPool, subKeyList);
            }
        }
        return poolKeysMap;
    }

redis集群配置

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.beans.factory.annotation.Qualifier;
import org.springframework.boot.autoconfigure.data.redis.RedisProperties;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.util.StringUtils;
import redis.clients.jedis.HostAndPort;
import redis.clients.jedis.JedisCluster;
import redis.clients.jedis.JedisPoolConfig;

import java.time.Duration;
import java.util.HashSet;
import java.util.Set;

/**
 * redis集群配置
 */
@Configuration
public class JedisClusterConfigure {

    /*@Value("#{'${redis.nodes}'.split(',')}")
    private List<String> nodes;*/

    @Autowired
    @Qualifier("spring.redis-org.springframework.boot.autoconfigure.data.redis.RedisProperties")
    private RedisProperties redisProperties;

    /**
     * @param
     * @desc: Redis 连接参数配置
     * @return:
     */
    @Bean
    public JedisPoolConfig jedisPoolConfig() {
        JedisPoolConfig jedisPoolConfig = new JedisPoolConfig();
        RedisProperties.Pool pool = redisProperties.getJedis().getPool();
        jedisPoolConfig.setMaxTotal(pool.getMaxActive());
        jedisPoolConfig.setMaxIdle(pool.getMaxIdle());
        jedisPoolConfig.setMinIdle(pool.getMinIdle());
        jedisPoolConfig.setMaxWaitMillis(pool.getMaxWait().toMillis());
        jedisPoolConfig.setTestOnBorrow(true);
        return jedisPoolConfig;
    }

    @Bean
    public JedisCluster jedisCluster() {
        Set<HostAndPort> nodes = new HashSet<>();
        for (String node : redisProperties.getCluster().getNodes()) {
            String[] hostAndPort = node.split(":");
            nodes.add(new HostAndPort(hostAndPort[0], Integer.parseInt(hostAndPort[1])));
        }
        Integer redirects = redisProperties.getCluster().getMaxRedirects();
        redirects = redirects != null ? redirects : 5;
        Duration duration = redisProperties.getTimeout();
        int timeout = Math.toIntExact(duration != null ? duration.toMillis() : 2000);
        String password = redisProperties.getPassword();
        return StringUtils.hasText(password)
                ? new JedisCluster(nodes, timeout, timeout, redirects, password, jedisPoolConfig())
                : new JedisCluster(nodes, timeout, timeout, redirects, jedisPoolConfig());
    }

}
