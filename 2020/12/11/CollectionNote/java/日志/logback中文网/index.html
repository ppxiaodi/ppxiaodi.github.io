logback.xml和logback-spring.xml的区别
logback和logback-spring.xml都可以用来配置logback，但是2者的加载顺序是不一样的。

logback.xml—>application.properties—>logback-spring.xml.

logback.xml加载早于application.properties，所以如果你在logback.xml使用了变量时，而恰好这个变量是写在application.properties时，那么就会获取不到，只要改成logback-spring.xml就可以解决。

例子：
logback-spring.xml

<?xml version="1.0" encoding="UTF-8"?>
<configuration debug="false">
	<!--定义日志文件的存储地址 勿在 LogBack 的配置中使用相对路径 -->
	<property name="LOG_HOME" value="logs"/>
	<!--读取配置中心的属性-->
	<springProperty scope="context" name="logging.kafka.servers" source="logging.elk.kafka.servers"
					defaultValue="x.x.x.x:9092"/>
	<springProperty scope="context" name="logging.kafka_audit.servers" source="logging.elk.kafka_audit.servers"
					defaultValue="x.x.x.x:9092"/>
	<springProperty scope="context" name="logging.kafka.topic" source="logging.elk.kafka.topic"
					defaultValue="topic-safecontrol"/>
	<springProperty scope="context" name="logging.kafka_audit.topic" source="logging.elk.kafka_audit.topic"
					defaultValue="topic-audit"/>
	<springProperty scope="context" name="logging.elk.level" source="logging.elk.level"
					defaultValue="ERROR"/>
	<springProperty scope="context" name="service.name" source="spring.application.name" defaultValue=""/>
	<springProperty scope="context" name="service.tags" source="spring.application.tags" defaultValue=""/>

	<!-- 控制台输出 -->
	<logger name="com.alibaba.nacos.client.naming" level="ERROR"/>
	<appender name="CONSOLE" class="ch.qos.logback.core.ConsoleAppender">
		<encoder class="ch.qos.logback.classic.encoder.PatternLayoutEncoder">
			<!--格式化输出：%d表示日期，%thread表示线程名，%-5level：级别从左显示5个字符宽度%msg：日志消息，%n是换行符 -->
			<pattern>%d{yyyy-MM-dd HH:mm:ss.SSS} [%thread] %-5level %logger{50} - %msg%n</pattern>
		</encoder>

	</appender>

	<appender name="KAFKA" class="com.bst.logging.logback.kafka.KafkaAppender">
		<topic>
			${logging.kafka.topic}
		</topic>
		<bootstrapServers>
			${logging.kafka.servers}
		</bootstrapServers>
		<service>
			${service.name}
		</service>
		<serviceTags>
			${service.tags}
		</serviceTags>
		<encoder class="ch.qos.logback.core.encoder.LayoutWrappingEncoder">
			<layout class="com.bst.logging.logback.kafka.KafkaJsonLayout"/>
		</encoder>
		<filter class="ch.qos.logback.classic.filter.ThresholdFilter">
			<level>${logging.elk.level}</level>
		</filter>
		<filter class="ch.qos.logback.core.filter.EvaluatorFilter">
			<evaluator name="kafkaClientLog">
				<expression>logger.equals("org.apache.kafka.clients.NetworkClient")</expression>
			</evaluator>
			<OnMatch>DENY</OnMatch>
			<OnMismatch>NEUTRAL</OnMismatch>
		</filter>
		<filter class="ch.qos.logback.core.filter.EvaluatorFilter">
			<evaluator name="AuditKafkaLog">
				<expression>logger.equals("safe_contrl_Audit")</expression>
			</evaluator>
			<OnMatch>DENY</OnMatch>
			<OnMismatch>NEUTRAL</OnMismatch>
		</filter>
	</appender>

	<appender name="ASYNC-KAFKA" class="ch.qos.logback.classic.AsyncAppender">
		<discardingThreshold>0</discardingThreshold>
		<queueSize>2048</queueSize>
		<neverBlock>true</neverBlock>
		<includeCallerData>true</includeCallerData>
		<appender-ref ref="KAFKA"/>
	</appender>

	<appender name="KAFKA_AUDIT" class="com.bst.logging.logback.kafka.KafkaAppender">
		<topic>
			${logging.kafka_audit.topic}
		</topic>
		<bootstrapServers>
			${logging.kafka_audit.servers}
		</bootstrapServers>
		<service>
			${service.name}
		</service>
		<serviceTags>
			${service.tags}
		</serviceTags>

		<encoder class="ch.qos.logback.core.encoder.LayoutWrappingEncoder">
			<layout class="com.bst.logging.logback.kafka.KafkaJsonLayout"/>
		</encoder>
		<filter class="ch.qos.logback.classic.filter.LevelFilter">
			<level>INFO</level>
		</filter>
		<filter class="ch.qos.logback.core.filter.EvaluatorFilter">
			<evaluator name="kafkaClientLog">
				<expression>logger.equals("org.apache.kafka.clients.NetworkClient")</expression>
			</evaluator>
			<OnMatch>DENY</OnMatch>
			<OnMismatch>NEUTRAL</OnMismatch>
		</filter>
		<filter class="ch.qos.logback.core.filter.EvaluatorFilter">
			<evaluator name="AuditKafkaLog">
				<expression>logger.equals("safe_contrl_Audit")</expression>
			</evaluator>
			<OnMatch>NEUTRAL</OnMatch>
			<OnMismatch>DENY</OnMismatch>
		</filter>
	</appender>

	<appender name="ASYNC-KAFKA_AUDIT" class="ch.qos.logback.classic.AsyncAppender">
		<discardingThreshold>0</discardingThreshold>
		<queueSize>2048</queueSize>
		<neverBlock>true</neverBlock>
		<includeCallerData>true</includeCallerData>
		<appender-ref ref="KAFKA_AUDIT"/>
	</appender>

	<!--<logger name="kafka-event" additivity="false">
        <appender-ref ref="ASYNC-KAFKA" />
    </logger>-->

	<!-- 按照每天生成日志文件 -->
	<appender name="FILE" class="ch.qos.logback.core.rolling.RollingFileAppender">
		<file>${LOG_HOME}/springboot.out</file>
		<rollingPolicy class="ch.qos.logback.core.rolling.TimeBasedRollingPolicy">
			<!--日志文件输出的文件名 -->
			<FileNamePattern>${LOG_HOME}/log/%d{yyyy-MM-dd}_info.log</FileNamePattern>
			<!--日志文件保留天数 -->
			<MaxHistory>30</MaxHistory>
		</rollingPolicy>
		<encoder class="ch.qos.logback.classic.encoder.PatternLayoutEncoder">
			<!--格式化输出：%d表示日期，%thread表示线程名，%-5level：级别从左显示5个字符宽度%msg：日志消息，%n是换行符 -->
			<pattern>%d{yyyy-MM-dd HH:mm:ss.SSS} [%thread] %-5level %logger{50} - %msg%n</pattern>
		</encoder>
		<!--日志文件最大的大小 -->
		<triggeringPolicy class="ch.qos.logback.core.rolling.SizeBasedTriggeringPolicy">
			<MaxFileSize>100MB</MaxFileSize>
		</triggeringPolicy>
	</appender>

	<appender name="ERROR_FILE" class="ch.qos.logback.core.rolling.RollingFileAppender">
		<file>${LOG_HOME}/error.log</file>
		<rollingPolicy class="ch.qos.logback.core.rolling.TimeBasedRollingPolicy">
			<fileNamePattern>${LOG_HOME}/error/%d{yyyy-MM-dd}_error.log</fileNamePattern>
			<maxHistory>90</maxHistory>
		</rollingPolicy>
		<triggeringPolicy class="ch.qos.logback.core.rolling.SizeBasedTriggeringPolicy">
			<!-- 文件大小触发重写新文件 -->
			<MaxFileSize>100MB</MaxFileSize>
			<!-- <totalSizeCap>10GB</totalSizeCap> -->
		</triggeringPolicy>
		<layout class="ch.qos.logback.classic.PatternLayout">
			<Pattern>%d{yyyy-MM-dd HH:mm:ss.SSS} [%thread] %-5level %logger{50} - %msg%n</Pattern>
		</layout>
		<filter class="ch.qos.logback.classic.filter.LevelFilter">
			<level>ERROR</level>
			<onMismatch>DENY</onMismatch>
		</filter>
	</appender>

	<appender name="ERROR_KAFKA_LOCAL" class="ch.qos.logback.core.rolling.RollingFileAppender">
		<file>${LOG_HOME}/elk.log</file>
		<rollingPolicy class="ch.qos.logback.core.rolling.TimeBasedRollingPolicy">
			<fileNamePattern>${LOG_HOME}/elk/%d{yyyy-MM-dd}_elk.log</fileNamePattern>
			<maxHistory>90</maxHistory>
		</rollingPolicy>
		<triggeringPolicy class="ch.qos.logback.core.rolling.SizeBasedTriggeringPolicy">
			<!-- 文件大小触发重写新文件 -->
			<MaxFileSize>100MB</MaxFileSize>
			<!-- <totalSizeCap>10GB</totalSizeCap> -->
		</triggeringPolicy>
		<layout class="ch.qos.logback.classic.PatternLayout">
			<Pattern>%msg%n</Pattern>
		</layout>
		<filter class="ch.qos.logback.classic.filter.LevelFilter">
			<level>INFO</level>
			<onMismatch>DENY</onMismatch>
		</filter>
	</appender>

	<appender name="ASYNC-LOCAL" class="ch.qos.logback.classic.AsyncAppender">
		<discardingThreshold>0</discardingThreshold>
		<queueSize>2048</queueSize>
		<appender-ref ref="ERROR_KAFKA_LOCAL"/>
	</appender>

	<!--万一kafka队列不通,记录到本地 -->
	<logger name="local" additivity="false">
		<appender-ref ref="ASYNC-LOCAL"/>
	</logger>

	<!-- 日志输出级别 -->
	<root level="INFO">
		<appender-ref ref="CONSOLE"/>
		<appender-ref ref="FILE"/>
		<appender-ref ref="ERROR_FILE"/>
		<appender-ref ref="ASYNC-KAFKA"/>
		<appender-ref ref="ASYNC-KAFKA_AUDIT"/>
	</root>

</configuration>
